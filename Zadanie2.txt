import os

nazwa = input("Podaj nazwę pliku lub katalogu: ")

//Część 1.1 zadania 2
if not os.path.exists(nazwa):
    print("Plik nie istnieje")

//Część 1.2 zadania 2
elif os.path.isdir(nazwa):
    print("Plik istnieje i jest katalogiem")

//Część 1.3 zadania 3
elif os.path.isfile(nazwa):
    print("Plik istnieje i jest plikiem zwykłym")
else:
    print("Plik istnieje, ale nie jest ani katalogiem, ani plikiem zwykłym")

//Część 2 zadania 2
import os

nazwa = input("Podaj nazwę zbioru: ")

//Część 2.1 zadania 2
if not os.path.exists(nazwa):
    print("Zbiór nie istnieje")

//Część 2.3 zadania 2
elif os.path.isdir(nazwa):
    print("Zbiór jest katalogiem")

//Część 2.4 zadania 2
elif not os.access(nazwa, os.R_OK):
    print("Zbiór nie można czytać")

//Część 2.2 zadania 2
elif os.path.isfile(nazwa):
    print("Zawartość pliku:")
    with open(nazwa, 'r', encoding='utf-8') as f:
        print(f.read())
else:
    print("Zbiór istnieje, ale nie jest ani plikiem zwykłym, ani katalogiem")

//Część 3 zadania 2
import os
import subprocess

nazwa = input("Podaj nazwę pliku do uruchomienia: ")

if os.path.isfile(nazwa) and os.access(nazwa, os.X_OK):
    # uruchomienie programu
    proces = subprocess.run([nazwa])
    print(f"Program zakończył się kodem wyjścia: {proces.returncode}")
else:
    print("Plik nie jest zwykłym plikiem wykonywalnym")
